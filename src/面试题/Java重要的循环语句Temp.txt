public class Java重要的循环语句 {
    public static void main(String[] args) {
        for(int i=0;i<10;i++){
            System.out.println(i);
        }
        int i=0;
        while (i<10){
            System.out.println(i);
            i++;
        }
        i=0;
        do{
            System.out.println(i);
            i++;
        }while (i<10);
    }
}
